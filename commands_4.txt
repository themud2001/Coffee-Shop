create database coffee_shop;

create table users (id INT(6) PRIMARY KEY AUTO_INCREMENT NOT NULL, username VARCHAR(20) NOT NULL, email VARCHAR(30) NOT NULL, phone VARCHAR(20) NOT NULL, password VARCHAR(255) NOT NULL);

create table products (id INT(6) PRIMARY KEY AUTO_INCREMENT NOT NULL, name VARCHAR(20) NOT NULL, price VARCHAR(10) NOT NULL, image VARCHAR(100) NOT NULL);

create table blogs (id INT PRIMARY KEY AUTO_INCREMENT NOT NULL, title VARCHAR(30) NOT NULL, description VARCHAR(100) NOT NULL, image VARCHAR(100) NOT NULL);

create table reviews (id INT PRIMARY KEY AUTO_INCREMENT NOT NULL, username VARCHAR(20) NOT NULL, email VARCHAR(30) NOT NULL, comment TEXT NOT NULL, rating INT(2) NOT NULL);

create table orders (id INT PRIMARY KEY AUTO_INCREMENT NOT NULL, total VARCHAR(20) NOT NULL, email VARCHAR(30) NOT NULL, timestamp VARCHAR(30) NOT NULL);

create table menu (id INT(6) PRIMARY KEY AUTO_INCREMENT NOT NULL, name VARCHAR(20) NOT NULL, price VARCHAR(10) NOT NULL, image VARCHAR(100) NOT NULL);

create table favorites (id INT AUTO_INCREMENT PRIMARY KEY NOT NULL, name VARCHAR(20) NOT NULL, price VARCHAR(10) NOT NULL, image VARCHAR(100) NOT NULL, email VARCHAR(30) NOT NULL);
